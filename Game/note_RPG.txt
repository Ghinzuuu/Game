- toString()
    Avec un StringBuilder -> organisation, lisibilité
    et on évite de créer des String supplémentaires.
    
    Pour le line feed, préférer System.lineSeparator()
    A stocker dans une classe utilitaire comme constance ?

	public String toString() {
		
		final String LINE_SEPARATOR = System.lineSeparator();

		StringBuilder builder = new StringBuilder();
		
		builder.append(type.name());
		builder.append(LINE_SEPARATOR);
		
		builder.append("Degat : ");
		builder.append(degat);
		builder.append(LINE_SEPARATOR);
		
		builder.append("Precision : ");
		builder.append(precision + "%");
		builder.append(LINE_SEPARATOR);
		
		builder.append("Critique : ");
		builder.append(critique + "%");
		builder.append(LINE_SEPARATOR);
		
		builder.append("Type : ");
		builder.append(type);
		builder.append(LINE_SEPARATOR);
		
		builder.append("Portee :");
		builder.append(portee);
		builder.append(LINE_SEPARATOR);
		
		builder.append(prix);
		builder.append(" PO");
		builder.append(LINE_SEPARATOR);
		
		return builder.toString();
	}
